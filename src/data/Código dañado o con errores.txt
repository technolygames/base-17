public void insertar_emp(String password,int codigo_emp,String nombre_emp,String apellidop_emp,String apellidom_emp,String puesto,String experiencia,String grado_estudios,int edad,FileInputStream imagen){
        String ins1_query="insert into empleados(password,codigo_emp,nombre_emp,apellidop_emp,apellidom_emp,puesto,experiencia,grado_estudios,edad,imagen,fecha_registro,fecha_sesion) values(?,?,?,?,?,?,?,?,?,?,now(),now());";
        try{
            ps=getConnection().prepareStatement(ins1_query);
            
            ps.setString(1,password);
            ps.setInt(2,codigo_emp);
            ps.setString(3,nombre_emp);
            ps.setString(4,apellidop_emp);
            ps.setString(5,apellidom_emp);
            ps.setString(6,puesto);
            ps.setString(7,experiencia);
            ps.setString(8,grado_estudios);
            ps.setInt(9,edad);
            ps.setBinaryStream(10,imagen);
            
            ps.execute();
            JOptionPane.showConfirmDialog(null,"Se han guardado los datos","Rel 1",JOptionPane.DEFAULT_OPTION);
        }catch(SQLException e){
            JOptionPane.showMessageDialog(null,"Error:\n"+e.getMessage(),"Error 11",JOptionPane.WARNING_MESSAGE);
            e.printStackTrace();
        }
    }
